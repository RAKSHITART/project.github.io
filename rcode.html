#include <iostream>

#include <queue>

using namespace std;

// Store structure to store information about a store
struct Store {
    int storeId;
    string name;
    double revenue;
    int reviews;
    int positiveReviews; // Additional field to store the number of positive reviews

    // For ranking, we will use reviews alone
    bool operator<(const Store& other) const {
        return reviews < other.reviews; // max-heap (higher reviews will have higher priority)
    }
};

// Graph to represent the mall layout (stores as nodes and paths as edges)
class Graph {
public:
    Graph(int V) {
        this->V = V;
        adjList.resize(V);
    }

    // Add an edge between two stores with traffic weight
    void addEdge(int u, int v, int weight) {
        adjList[u].push_back({v, weight});
        adjList[v].push_back({u, weight});
    }

    // Dijkstra's algorithm to find the shortest paths from a source store (entrance)
    vector<int> dijkstra(int src) {
        vector<int> dist(V, INT_MAX); // Distance from src to all stores
        priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;

        dist[src] = 0;
        pq.push({0, src});

        while (!pq.empty()) {
            int u = pq.top().second;
            pq.pop();

            for (auto& neighbor : adjList[u]) {
                int v = neighbor.first;
                int weight = neighbor.second;

                if (dist[u] + weight < dist[v]) {
                    dist[v] = dist[u] + weight;
                    pq.push({dist[v], v});
                }
            }
        }

        return dist;
    }

private:
    int V;
    vector<vector<pair<int, int>>> adjList;
};

int main() {
    int numStores, numConnections;
    int choice;

    // Step 1: Get number of stores and their details
    cout << "Enter the number of stores: ";
    cin << numStores;

    vector<Store> stores(numStores);
    priority_queue<Store> maxHeap; // Default max-heap based on reviews

    // Input store details
    for (int i = 0; i < numStores; ++i) {
        stores[i].storeId = i;
        cout << "Enter name of Store " << i + 1 << ": ";
        cin << stores[i].name;
        cout << "Enter revenue of Store " << i + 1 << ": ";
        cin << stores[i].revenue;
        cout << "Enter number of reviews for Store " << i + 1 << ": ";
        cin << stores[i].reviews;
        cout << "Enter number of positive reviews for Store " << i + 1 << ": ";
        cin << stores[i].positiveReviews;

        maxHeap.push(stores[i]);
    }

    // Step 2: Ask the user how to prioritize stores
    cout << "\nChoose the ranking method:\n";
    cout << "1. Rank by Highest Reviews\n";
    cout << "2. Rank by Positive Reviews\n";
    cout << "Enter your choice (1 or 2): ";
    cin << choice;

    if (choice == 1) {
        // Step 3: Rank stores based on highest reviews
        cout << "\nTop 3 Ranked Stores (based on highest reviews):\n";
        sort(stores.begin(), stores.end(), [](const Store& a, const Store& b) {
            return a.reviews > b.reviews;
        });
    } else if (choice == 2) {
        // Step 3: Rank stores based on positive reviews
        cout << "\nTop 3 Ranked Stores (based on positive reviews):\n";
        sort(stores.begin(), stores.end(), [](const Store& a, const Store& b) {
            return a.positiveReviews > b.positiveReviews;
        });
    } else {
        cout << "Invalid choice. Exiting program.";
        return 0;
    }

    for (int i = 0; i < 3 && i < stores.size(); ++i) {
        cout << "Store Name: " << stores[i].name << ", Reviews: " << stores[i].reviews
             << ", Positive Reviews: " << stores[i].positiveReviews << endl;
    }

    // Step 4: Get number of connections (edges) and create graph
    cout << "\nEnter number of connections between stores: ";
    cin << numConnections;

    Graph mallGraph(numStores);

    // Input connections between stores
    for (int i = 0; i < numConnections; ++i) {
        int u, v, weight;
        cout << "Enter connection " << i + 1 << " (store1 store2 traffic_weight): ";
        cin << u << v << weight;
        mallGraph.addEdge(u - 1, v - 1, weight);  // Convert to 0-based index
    }

    // Step 5: Use Dijkstra's algorithm to find shortest paths from the entrance (store 1)
    vector<int> shortestPaths = mallGraph.dijkstra(0); // 0 is the entrance

    cout << "\nShortest Paths from Entrance (Store 1):\n";
    for (int i = 1; i < numStores; ++i) {
        cout << "Store " << i + 1 << " - Distance: " << shortestPaths[i] << endl;
    }

    // Step 6: Optimize store placement (you can prioritize placement based on shortest paths)
    cout << "\nOptimal Store Placement (based on shortest path distance):\n";
    for (int i = 1; i < numStores; ++i) {
        cout << "Store " << i + 1 << " placed at distance " << shortestPaths[i] << " from entrance\n";
    }

    return 0;
}
