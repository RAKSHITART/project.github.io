2. Revenue Prediction for Shops

Scenario: The goal is to analyze the revenue trends of shops to predict their future performance based on historical data and inter-shop relationships.

Algorithm: Merge Sort and Depth First Search (DFS)

Approach:

  • Input Format: Revenue data as pairs {revenue, shopID} and shop relationships as pairs of connected shop IDs.
  • Merge Sort: Sorts revenue data by revenue amounts or chronologically (depending on input).
  • DFS: Explores the relationships between shops based on the graph input.

Output:

  • Displays sorted revenue data.
  • Analyzes and outputs shop relationships starting from each unvisited shop.
#include <iostream>
#include <vector>
#include <algorithm>
#include <map>
#include <stack>

using namespace std;

// Merge function for Merge Sort
void merge(vector<pair<int, int>> &revenues, int left, int mid, int right) {
    int n1 = mid - left + 1;
    int n2 = right - mid;

    vector<pair<int, int>> L(n1), R(n2);

    for (int i = 0; i < n1; ++i)
        L[i] = revenues[left + i];
    for (int j = 0; j < n2; ++j)
        R[j] = revenues[mid + 1 + j];

    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2) {
        if (L[i].first <= R[j].first) {
            revenues[k++] = L[i++];
        } else {
            revenues[k++] = R[j++];
        }
    }

    while (i < n1) revenues[k++] = L[i++];
    while (j < n2) revenues[k++] = R[j++];
}

// Merge Sort
void mergeSort(vector<pair<int, int>> &revenues, int left, int right) {
    if (left < right) {
        int mid = left + (right - left) / 2;
        mergeSort(revenues, left, mid);
        mergeSort(revenues, mid + 1, right);
        merge(revenues, left, mid, right);
    }
}

// DFS for analyzing shop relationships
void dfs(int node, map<int, vector<int>> &graph, vector<bool> &visited) {
    stack<int> stack;
    stack.push(node);

    cout << "Analyzing relationships starting from Shop " << node << "...\n";

    while (!stack.empty()) {
        int current = stack.top();
        stack.pop();

        if (!visited[current]) {
            visited[current] = true;
            cout << "Shop " << current << " is connected to: ";
            for (int neighbor : graph[current]) {
                cout << neighbor << " ";
                if (!visited[neighbor]) {
                    stack.push(neighbor);
                }
            }
            cout << "\n";
        }
    }
}

int main() {
    int n, m;

    // Input for revenue data
    cout << "Enter the number of revenue entries: ";
    cin >> n;

    vector<pair<int, int>> revenues(n); // {revenue, shop ID}
    cout << "Enter revenue data as: revenue shopID\n";
    for (int i = 0; i < n; ++i) {
        cin >> revenues[i].first >> revenues[i].second;
    }

    // Input for shop relationships
    cout << "Enter the number of relationships between shops: ";
    cin >> m;

    map<int, vector<int>> graph;
    cout << "Enter relationships as: shop1 shop2\n";
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        graph[u].push_back(v);
        graph[v].push_back(u);
    }

    // Sorting revenue data
    mergeSort(revenues, 0, n - 1);

    // Output sorted revenue data
    cout << "\n--- Revenue Prediction for Shops ---\n";
    cout << "Objective: Predict future performance by analyzing historical data and shop relationships.\n";
    cout << "Sorted Revenue Data (Chronologically or by Amount):\n";
    for (auto &entry : revenues) {
        cout << "Revenue: " << entry.first << ", Shop ID: " << entry.second << "\n";
    }

    // Analyzing shop relationships using DFS
    vector<bool> visited(graph.size(), false);
    for (const auto &pair : graph) {
        if (!visited[pair.first]) {
            dfs(pair.first, graph, visited);
        }
    }

    return 0;
}
